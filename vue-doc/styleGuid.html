# CSS

遵循 BEM 的命名：block-name__element-name--modifier-name
也就是(Block)模块名 + (Element)元素名 + (modifier)修饰器名，元素名用"--"连接，修饰器名用"--"连接。

<template>
  <button
    :class="[b(['', size, type]),
      { 'is-disabled': disabled || loading },
      { 'is-round' : round }
    ]"
  >
  </button>
</template>

<script lang="ts">
import { defineComponent } from 'vue'
import prefix, { PREFIX } from '../../mixins/prefix'

// bem helper
// b() // 'gw-button'
// b('text') // 'gw-button__text'
// b({ disabled }) // 'gw-button button--disabled'
// b('text', { disabled }) // 'gw-button__text gw-button__text--disabled'
// b(['disabled', 'primary']) // 'gw-button gw-button--disabled gw-button--primary'
export default defineComponent({
  name: `${PREFIX}-button`,
  mixins: [prefix],
})
</script>
