# 开发指南

# 一、 准备 node 环境大于 v12

# 二、 全局 安装 yarn

npm i -g yarn
1

# 三、 安装依赖及初始化 lerna

# 该命令包含安装和初始化 yarn && yarn lerna bootstrap
yarn bootstrap
1
2

如果上述命令中的 yarn 不能安装依赖，可以使用 npm 安装，安装前请先删除package-lock.json 和 yarn.lock, 之后的命令也都使用 npm

# 第一步 安装依赖
npm i

# 第二步 初始化lerna，先把 lerna.json 中的 npmClient 改成 npm
npm run lerna bootstrap

开发目录为 packages，公共模块有mixins、styles、utils

以开发Button组件为例，新建 Button 文件，下面包含

  • _tests_ 测试文件目录
  • demo 文档例子目录
  • src 组件文件目录
  • index.ts 组件入口文件
  • package.json 组件的 package.json

# 八、命令含义

"scripts": {
  "bootstrap": "yarn && yarn lerna bootstrap",  // 初始化
  "gen": "bash ./scripts/gc.sh", // 新建一个组件文件命令
  "dev": "concurrently \"yarn docs:dev\" \"yarn demo:serve\" ", // 运行文档和demo
  "demo:serve": "vue-cli-service serve", // 运行demo
  "demo:build": "vue-cli-service build", // 打包demo
  "demo:lint": "vue-cli-service lint", // 校验demo
  "docs:dev": "vuepress dev docs", // 运行文档
  "docs:build": "vuepress build docs", // 打包文档
  "build": "concurrently \"yarn build:components\" \"yarn build:bundle\" ", // 打包组件和bundle
  "build:components": "yarn rollup -c build/rollup.config.js", // 打包组件
  "build:bundle": "yarn rollup -c build/rollup.bundle.config.js", // 打包组件的bundle
  "css:lint": "npx stylelint '**/*.scss' --fix", // 校验样式并修复错误
  "lint": "eslint 'packages/**/*.{js,ts,vue}' --fix", // eslint
  "test": "jest" // 单元测试
},
