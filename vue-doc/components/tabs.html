# 基本用法

通过 v-model:active 绑定当前激活标签对应的索引值，默认情况下启用第一个标签, 下标为索引时从 0 开始

<gw-tabs v-model:active="active">
  <gw-tab title="标签 1">内容 1</gw-tab>
  <gw-tab title="标签 2">内容 2</gw-tab>
  <gw-tab title="标签 3">内容 3</gw-tab>
  <gw-tab title="标签 4">内容 4</gw-tab>
</gw-tabs>

# 匹配名称

在标签指定 name 属性的情况下，v-model:active 的值为当前标签的 name（此时无法通过索引值来匹配标签）。

<gw-tabs v-model:active="activeName">
  <gw-tab title="标签 1" name="a">内容 1</gw-tab>
  <gw-tab title="标签 2" name="b">内容 2</gw-tab>
  <gw-tab title="标签 3" name="c">内容 3</gw-tab>
</gw-tabs>

# 禁用标签

设置 disabled 属性即可禁用标签，如果需要监听禁用标签的点击事件，可以在 gw-tabs 上监听 disabled 事件。

<gw-tabs>
  <gw-tab title="标签 1">内容 1</gw-tab>
  <gw-tab title="标签 2">内容 2</gw-tab>
  <gw-tab title="标签 3" disabled>内容 3</gw-tab>
  <gw-tab title="标签 4">内容 4</gw-tab>
</gw-tabs>

# 样式风格

Tab 支持两种样式风格：line 和 card，默认为 line 样式，可以通过 type 属性切换样式风格。

<gw-tabs type="card">
  <gw-tab title="标签 1">内容 1</gw-tab>
  <gw-tab title="标签 2">内容 2</gw-tab>
  <gw-tab title="标签 3" disabled>内容 3</gw-tab>
  <gw-tab title="标签 4">内容 4</gw-tab>
</gw-tabs>

# 点击事件

可以在 gw-tabs 上绑定 click 事件，事件传参为标签对应的标识符和标题。

<gw-tabs @click="onClick" type="card">
  <gw-tab title="标签 1">内容 1</gw-tab>
  <gw-tab title="标签 2">内容 2</gw-tab>
  <gw-tab title="标签 3">内容 3</gw-tab>
</gw-tabs>

# 异步事件

beforeChange 属性可以在切换之前做一些特别的操作，来判断是否切换

<gw-tabs type="card" :before-change="beforeChange">
  <gw-tab title="标签 1">内容 1</gw-tab>
  <gw-tab title="标签 2">内容 2</gw-tab>
  <gw-tab title="标签 3">内容 3</gw-tab>
</gw-tabs>

setup(){
  const beforeChange = (index: number) => {
    // 返回 false 表示阻止此次切换
    if (index === 1) {
      console.log('阻止此次切换')
      return false;
    }
  };
  return { beforeChange }
}
